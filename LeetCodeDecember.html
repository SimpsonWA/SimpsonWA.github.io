Problem: 1475. Final Prices With a Special Discount in a Shop

Solution:


class Solution {
public:
    vector finalPrices(vector& prices) {
        vector ans = prices;
        for (int i = 0; i < prices.size(); i++) {
            for (int j = i + 1; j < prices.size(); j++) {
                if (prices[j] <= prices[i]) {
                    ans[i] = prices[i] - prices[j];
                    break;
                }
            }
        }
        return ans;
    }
};

Explanation:

  • Quick and dirty O(nÂ²) solution: Loop through each index. At each starting index, iterate through until prices[j] <= prices[i].
