Problem: 1475. Final Prices With a Special Discount in a Shop

Solution:


class Solution {
public:
    vector finalPrices(vector& prices) {
        vector ans = prices;
        for (int i = 0; i < prices.size(); i++) {
            for (int j = i + 1; j < prices.size(); j++) {
                if (prices[j] <= prices[i]) {
                    ans[i] = prices[i] - prices[j];
                    break;
                }
            }
        }
        return ans;
    }
};

Explanation:

  • Quick and dirty O(nÂ²) solution: Loop through each index. At each starting index, iterate through until prices[j] <= prices[i].

Problem: 769. Max Chunks To Make Sorted

Solution:

                    
class Solution {
public:
    int maxChunksToSorted(vector& arr) {
        int count = 0, diff = 0;
                                    
        for(int i = 0; i < arr.size(); i++) {
            diff += arr[i] - i;
            count += (diff == 0);
        }
        
        return count;
    }
};

Explanation:

  • Essentally just need to calculate the difference between the current element and its index and track when this difference is zero, which indicates that a chunk can be sorted independently.

Problem: 2415. Reverse Odd Levels of Binary Tree

Solution:

                    
class Solution {
public:
    TreeNode* reverseOddLevels(TreeNode* root) {
        dfs(root->left, root->right, 1);
        return root;
    }
private:
    void dfs(TreeNode* leftNode, TreeNode* rightNode, int level){
        if(!leftNode || !rightNode) {
            return;
            } 
                            
        if(level % 2){
            swap(leftNode->val, rightNode->val);
            } 
                            
        dfs(leftNode->left, rightNode->right, level+1);
        dfs(leftNode->right, rightNode->left, level+1);
    }
};
